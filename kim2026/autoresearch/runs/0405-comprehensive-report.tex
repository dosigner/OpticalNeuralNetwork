\begin{itemize}
\item \textbf{D2NN}: 5 phase-only layers, spacing 10mm, detector 10mm, pad factor 2
\item \textbf{Optics}: $\lambda$=1.55$\mu$m, receiver window=2.048mm, aperture=2mm

\item \textbf{Atmosphere}: Cn$^2$=5$\times$10$^{-14}$ m$^{-2/3}$, distances 100m--3km

\item \textbf{Training}: 30 epochs, batch=32, LR=8$\times$10$^{-3}$$\rightarrow$cosine, TV=0.05, A100
\end{itemize}

\section{Received Power Gap Decomposition}

\begin{align*}
\text{Bucket Power} &\approx \underbrace{\text{TP}}_{\text{м—җл„Ҳм§Җ ліҙмЎҙ}} \times \underbrace{\exp(-\sigma^2_{\text{WFE}})}_{\text{Strehl}} \times P_{\text{diffraction-limited}} \\[0.5em]
\text{WFE} &= 442.9\text{nm} = 1.80\text{ rad} \quad \Rightarrow \quad \text{Strehl} = e^{-\sigma^2} = 0.04
\end{align*}

\section{Conclusions}

\begin{enumerate}
\item \textbf{Normalized PIB $\neq$ мҲҳмӢ м „л Ҙ.} PIB 90.1\%м—¬лҸ„ м ҲлҢҖ м „л ҘмқҖ turbulentмқҳ 59\%.
    Throughputмқ„ мң м§Җн•ҳлҠ” мҶҗмӢӨн•ЁмҲҳл§Ң мӢӨм „ мң нҡЁ.

\item \textbf{Raw RP lossк°Җ мөңм Ғ}: $\mathcal{L} = -\log(\text{bucket}/\text{input})$.
    Vacuum reference м—Ҷмқҙ м ҲлҢҖ мҲҳмӢ м „л Ҙмқ„ м§Ғм ‘ мөңлҢҖнҷ”.

\item \textbf{D2NN = beam shaper + statistical corrector.}
    Vacuumм—җлҸ„ нҡЁкіј мһҲмқҢ (Airy sidelobe redistribution).
    к°•н•ң лӮңлҘҳм—җм„ң нҶөкі„м Ғ ліҙм • нҡЁкіј м¶”к°Җ.

\item \textbf{кұ°лҰ¬к°Җ кёёмҲҳлЎқ D2NN мқҙл“қ мҰқк°Җ}: 3kmм—җм„ң +5.1 dB, deep fade +8.1 dB к°ңм„ .
    D/r$_0$ > 7м—җм„ң кёүмҰқ.

\item \textbf{Static D2NNмқҖ operating condition м „мҡ©}: 3km лӘЁлҚёмқ„ 500mм—җ м Ғмҡ©н•ҳл©ҙ $-$60\%.
    л°°м№ҳ нҷҳкІҪм—җ л§һм¶Ө м„Өкі„ н•„мҲҳ.

\item \textbf{мөңм Ғ мҙҲм җкұ°лҰ¬}: f=6.5mm (SMF-28 MFD=10.4$\mu$m л§Өм№ӯ).
    f=4.5mmм—җм„ң 13\% м»Өн”Ңл§Ғ мҶҗмӢӨ.

\item \textbf{Absolute bucket loss мҲҳмӢқ}:
    \[
    \mathcal{L}_{\text{raw-RP}} = -\log\!\left(\frac{\sum_{r \leq 10\mu\text{m}} |E_{\text{pred}}^{\text{focal}}|^2}
    {\sum |E_{\text{input}}|^2} + \epsilon\right)
    \]
\end{enumerate}
